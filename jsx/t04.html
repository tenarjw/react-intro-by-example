Poza props można użyć innych zmiennych lokalnych.
Zwróć uwagę na sposób zapisywania funkcji (nie można pominąć {}): 
let Hello = (props) => {
  let data=(new Date()).toDateString();
  return(
   <div>
    <h1>Hello {props.toWho}!!!!</h1>
    <p>Data: {data}</p>
   </div>); 
};
let element=<Hello toWho='World' />;
let div=document.getElementById('root');
ReactDOM.render(element,div);
